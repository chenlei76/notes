### 屁股线
    f(x) = x**x when x>0 and (-x)**(-x) when x<0


```python
def f(x):
    y = np.ones(x.shape)
    i = x>0
    y[i] = np.power(x[i],x[i])
    i = x<0
    y[i] = np.power(-x[i],-x[i])
    return y

x = np.linspace(-1.3,1.3,101)
y = f(x)
plt.plot(x,y,'g-',label='x^x',linewidth = 2)
plt.grid()
plt.legend(loc='upper left')
plt.show()
```
